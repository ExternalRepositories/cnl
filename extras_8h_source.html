55  namespace _impl {
56  template<class Rep>
57  CNL_NODISCARD constexpr Rep sqrt_solve3(Rep n, Rep bit, Rep result)
58  {
59  return (bit != Rep{0})
60  ? (n >= result + bit) ? sqrt_solve3<Rep>(
61  static_cast<Rep>(n - (result + bit)), static_cast<Rep>(bit >> 2),
62  static_cast<Rep>((result >> 1) + bit))
63  : sqrt_solve3<Rep>(
64  n, static_cast<Rep>(bit >> 2),
65  static_cast<Rep>(result >> 1))
66  : result;
67  }

69  template<int Exponent>
70  struct sqrt_solve1 {
71  template<class Rep>
72  CNL_NODISCARD constexpr Rep operator()(Rep n) const
73  {
74  using widened_rep = _impl::set_width_t<Rep, _impl::width<Rep>::value * 2>;
75  return static_cast<Rep>(sqrt_solve3<widened_rep>(
76  _impl::fixed_width_scale<-Exponent>(static_cast<widened_rep>(n)),
77  widened_rep(
78  (widened_rep{1} << ((countr_used(n) + 1 - Exponent) & ~1)) >> 2),
79  widened_rep{0}));
80  }
81  };

100  template<typename Rep, int Exponent, int Radix>
101  CNL_NODISCARD constexpr auto sqrt(scaled_integer<Rep, power<Exponent, Radix>> const& x)
102  -> scaled_integer<Rep, power<Exponent, Radix>>
103  {
104  using type = scaled_integer<Rep, power<Exponent, Radix>>;
105  return _impl::to_rep(x) < 0
106  ? _impl::unreachable<type>("negative value passed to cnl::sqrt")
107  : type{_impl::from_rep<type>(_impl::sqrt_solve1<Exponent>{}(unwrap(x)))};
108  }
